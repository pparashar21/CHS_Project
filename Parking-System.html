Source Code - Arduino

#include <ESP8266WiFi.h>
#include <FirebaseArduino.h>
#include <ThingSpeak.h>
// Set these to run example.
#define FIREBASE_HOST "example.firebaseio.com"
#define FIREBASE_AUTH "token_or_secret"
#define WIFI_SSID "SHR_4G"
#define WIFI_PASSWORD "shikh@0123"
WiFiClient client;
int Vehicle_Count, capacity;
const int EntranceTrigger = D5;// entrance sensor pin
const int EntranceEcho = D6; // entrance sensor pin
const int ExitTrigger = D7; // exit sensor pin
const int ExitEcho = D8; // exit sensor pin
double Threshold = 10; // Threshold distance in cm for detection
unsigned long Channel_ID = 1699162;
unsigned long myChannelNumber = 1;
const char* myWriteAPIKey = "JAQVL92ZU3250BQT";
String server = "api.thingspeak.com";
void setup() {
Serial.begin(115200);

// Entrance Setup
pinMode(EntranceEcho, INPUT);
pinMode(EntranceTrigger, OUTPUT);
// Exit Setup
pinMode(ExitTrigger, OUTPUT);
pinMode(ExitEcho, INPUT);
Vehicle_Count = 0;
capacity = 10;

// connect to wifi.
WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
Serial.print("connecting");
while (WiFi.status() != WL_CONNECTED) {
Serial.print(".");
delay(500);
}
Serial.println();
Serial.print("connected: ");
Serial.println(WiFi.localIP());

WiFi.mode(WIFI_STA);
ThingSpeak.begin(client);
//Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
}
int n = 0;
void loop() {
if(Entry()){
// an entry is detected
if(Vehicle_Count == capacity){
Serial.println("Parking space is full!");
delay(300);
}
else{
Vehicle_Count++;
Serial.println("A Vehicle is detected at Entrance");
Serial.println("Currently, There are " + (String)Vehicle_Count + " Vehicles in the Parking Area");
Serial.println("Parking Space available: " + (String)(capacity - Vehicle_Count));
Serial.println("-----------------------------------------------------------------\n\n");
Serial.println();
}
}
delay(500);
if(Exit()){
// an exit is detected
Vehicle_Count--;
Serial.println("A Vahicle is detected at Exit");
Serial.println("Currently, There are " + (String)Vehicle_Count + " Vehicles in the Parking Area");
Serial.println("Parking Space available: " + (String)(capacity - Vehicle_Count));
Serial.println("-----------------------------------------------------------------\n\n");
}
int x = ThingSpeak.writeField(Channel_ID, 1, String(Vehicle_Count), myWriteAPIKey);
if(x==200) Serial.println("Updated successfully");
else Serial.println("failed");
delay(1000);

Serial.println("Waiting...");
delay(10000);
}
bool Entry(){
digitalWrite(EntranceTrigger, LOW);
delayMicroseconds(2);
digitalWrite(EntranceTrigger, HIGH);
delayMicroseconds(10);
digitalWrite(EntranceTrigger, LOW);
long duration = pulseIn(EntranceEcho, HIGH);
int distance = duration * 0.034 / 2;
//Serial.print("entry distance:");
//Serial.println(distance);
if(distance < Threshold){
return true;
}
else{
return false;
}
}
bool Exit(){
digitalWrite(ExitTrigger, LOW);
delayMicroseconds(2);
digitalWrite(ExitTrigger, HIGH);
delayMicroseconds(10);
digitalWrite(ExitTrigger, LOW);
long duration = pulseIn(ExitEcho, HIGH);
int distance = duration * 0.034 / 2;
//Serial.print("exit distance:");
//Serial.println(distance);
if(distance < Threshold){
return true;
}
else{
return false;
}
}
